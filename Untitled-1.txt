            function toggleFields(fieldsToShow, fieldsToHide) {
                // Show required fields

            // Function to determine which fields to show/hide based on selections
            function updateFieldVisibility() {
                const selectedTypes = $('#property_type').val() || [];
                const selectedSubtypes = $('#property_subtype').val() || [];

                let allFieldsToShow = new Set();
                let fieldsToHide = new Set();

                // Always show basic fields
                const basicFields = ['price_min', 'price_max', 'street_name', 'public_remarks', 'state', 'listing_status', 'property_condition'];
                basicFields.forEach(field => allFieldsToShow.add(field));

                // Initialize all conditional fields as potentially hideable
                const conditionalFields = ['bedrooms_total_min', 'bedrooms_total_max', 'baths_total_min', 'baths_total_max',
                    'finished_sqft_total_min', 'finished_sqft_total_max', 'acres_min', 'acres_max',
                    'water_source', 'sewer'];

                // If no types or subtypes selected, show all fields
                if (selectedTypes.length === 0 && selectedSubtypes.length === 0) {
                    conditionalFields.forEach(field => allFieldsToShow.add(field));
                    toggleFields(Array.from(allFieldsToShow), []);
                    return;
                }

                // Determine what property categories we're dealing with
                let hasResidential = false;
                let hasLand = false;
                let hasCommercial = false;
                let hasBusinessOpp = false;

                // Check selected property types
                selectedTypes.forEach(type => {
                    const typeKey = type.toLowerCase();
                    switch (typeKey) {
                        case 'residential':
                        case 'residential income':
                        case 'residential lease':
                            hasResidential = true;
                            break;
                        case 'land':
                            hasLand = true;
                            break;
                        case 'commercial lease':
                        case 'commercial sale':
                            hasCommercial = true;
                            break;
                        case 'business opportunity':
                            hasBusinessOpp = true;
                            break;
                    }
                });

                // Check selected subtypes to determine categories
                selectedSubtypes.forEach(subtype => {
                    const subtypeKey = subtype.toLowerCase();
                    if (subtypeRules[subtypeKey]) {
                        const relatedType = subtypeRules[subtypeKey].type;
                        switch (relatedType) {
                            case 'residential':
                                hasResidential = true;
                                break;
                            case 'land':
                                hasLand = true;
                                break;
                            case 'commercial':
                                hasCommercial = true;
                                break;
                        }
                    }
                });

                // Build fields to show based on what categories are selected

                // Residential fields (bedrooms, bathrooms, square footage)
                if (hasResidential) {
                    ['bedrooms_total_min', 'bedrooms_total_max', 'baths_total_min', 'baths_total_max',
                        'finished_sqft_total_min', 'finished_sqft_total_max', 'water_source', 'sewer'].forEach(field => {
                            allFieldsToShow.add(field);
                        });
                }

                // Land fields (acres, utilities)
                if (hasLand) {
                    ['acres_min', 'acres_max', 'water_source', 'sewer'].forEach(field => {
                        allFieldsToShow.add(field);
                    });
                }

                // Commercial fields (square footage, acres, utilities)
                if (hasCommercial) {
                    ['finished_sqft_total_min', 'finished_sqft_total_max', 'acres_min', 'acres_max',
                        'water_source', 'sewer'].forEach(field => {
                            allFieldsToShow.add(field);
                        });
                }

                // Business opportunity fields (square footage, acres)
                if (hasBusinessOpp) {
                    ['finished_sqft_total_min', 'finished_sqft_total_max', 'acres_min', 'acres_max'].forEach(field => {
                        allFieldsToShow.add(field);
                    });
                }

                // Pure land selections should never show bedroom/bathroom fields
                if (hasLand && !hasResidential) {
                    ['bedrooms_total_min', 'bedrooms_total_max', 'baths_total_min', 'baths_total_max'].forEach(field => {
                        allFieldsToShow.delete(field);
                    });
                }

                // Pure commercial/business should not show bedroom/bathroom fields unless residential is also selected
                if ((hasCommercial || hasBusinessOpp) && !hasResidential) {
                    ['bedrooms_total_min', 'bedrooms_total_max', 'baths_total_min', 'baths_total_max'].forEach(field => {
                        allFieldsToShow.delete(field);
                    });
                }

                // Determine which fields to hide (conditional fields not in show list)
                conditionalFields.forEach(field => {
                    if (!allFieldsToShow.has(field)) {
                        fieldsToHide.add(field);
                    }
                });

                // Apply the changes
                toggleFields(Array.from(allFieldsToShow), Array.from(fieldsToHide));
            }

            // Event listeners
            $('#property_type').on('change', updateFieldVisibility);
            $('#property_subtype').on('change', updateFieldVisibility);

            // Initial call to set up form
            updateFieldVisibility();
        });

                'baths_total_max',
                'water_source',
                'sewer',
                'street_name',
                'public_remarks'
            ];

            fields.forEach(id => {
                let value = $('#' + id).val();

                if (Array.isArray(value) && value.length > 0) {
                    filters.push(`${id}:${value.join(',')}`);
                }
                else if (typeof value === 'string' && value.trim() !== '') {
                    // Encode string values for URLs
                    if (id === 'street_name' || id === 'public_remarks') {
                        value = encodeURIComponent(value);
                    }
                    filters.push(`${id}:${value}`);
                }
            });


            // Sorting & limit
            filters.push('limit:12');
            filters.push('order:price');

            // Redirect in a new tab
            const finalUrl = baseUrl + filters.join('+');
            window.open(finalUrl, '_blank');
        }
    </script>
